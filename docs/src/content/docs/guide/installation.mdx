import {Tabs, TabItem, LinkCard} from "@astrojs/starlight/components";

To install bebopc globally, run the following command:

<Tabs>
  <TabItem label="bash/zsh">
  ```bash
  bash -c "$(curl https://bebop.sh)"
  ```
  </TabItem>

  <TabItem label="powershell">
  ```powershell
  irm https://bebop.sh | iex
  ```
  </TabItem>

</Tabs>


Alternatively, you can install it locally to your project:
<Tabs>
  <TabItem label="npm">
  ```bash
  npm add bebop-tools
  ```
  </TabItem>

  <TabItem label="yarn">
  ```bash
  yarn add bebop-tools
  ```
  </TabItem>

  <TabItem label="nuget">
  ```bash
  dotnet add package bebop-tools
  ```
  </TabItem>

  <TabItem label="cargo">
  ```bash
  cargo add bebop-tools
  ```
  </TabItem>

</Tabs>
